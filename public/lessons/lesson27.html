دائرةُ المُشْتَبَهِ 2

سببُ التّسميةِ: 3

سُمِّيَتْ بذلك للسّببينِ الآتيينِ: 4

1- تشابُهِ بعضِ أجزائِها حيثُ يشْتبِهُ "مُسْتَفْعِلُن" مجموعةُ الوتدِ "علن //5" بـ"مستفع لن" مفروقةِ الوتدِ "تف ع /5 /"، و"فَاعِلاتُنْ " مجموعةُ الوتدِ "علا //5" بـ "فاع لاتن" مفروقة الوتد "فاع/5/". 7

2- تشابُهِ أبـْحُرِها؛ لأنَّ "مُسْتَفْعِ لُن" في الخفيفِ والمجتثِّ مفروقُ الوتَدِ، وفي غيرهِما مجموعُ الوتِدِ "مستفعلن"، و"فَاعِ لاتُنْ" في المضارعِ مفروقُ الوتدِ، وفي غيرِهِ مجموعُ الوتد "فاعلاتن"، وقد أدَّى تشابُهِ الأبـْحر ِإلى التِباسِهَا على بعضِ الشُّعراءِ، فخَلطوا أبـْحُرَها في القصيدةِ الواحدةِ، وقَدْ حَكى "ابنُ القَّطاعِ" أنَّ فُحولَ الشُّعراء غَلِطوا في أبـحرِها، فأدخَلوا بعضَها على بَعضٍ توهماً مِنهُم أنَّهُ بحرٌ واحدٌ، مِنهمْ: "مهلهلٌ"، و"مرقَّشٌ"، و"عبيدُ بنُ الأبْرصِ"، و"علقمةُ بنُ عبدةَ"، ووقعَ مِن ذلكَ قصيدةٌ لـ"لطّرماحِ" حكاها "المعرِّي". 15

خصائصُها: 17

1- مُسدَّسةُ الأجزاءِ. 18

2- مَبنيَّةٌ على بحرِ السَّريعِ، فهوَ أصلُها رغمَ ابتدائِهِ بسببٍ، وكانَ قياسُهُم يقتضي بناءَها على المضارعِ؛ لابتدائِهِ بوتدٍ، إلاَّ أنَّـهم استبعدوهُ لأمرين: 21

الأوَّلُ: قِلَّتُهُ في الشِّعرِ حَتى أنْكرَهُ بعضُ اللُّغويينَ، والعَروضييِّنَ كـ"الزَّجاجِ"، ومِن ثَــمَّ فَهُوَ في حكمِ الـمُهملِ نظرًا لقلَّتِهِ، والمهملُ لا يُـجْعلُ أصلاً، ومِن ثَمَّ قدَّموا الأكثرَ استعمالاً. 3

الثّاني: عدمُ مجيءِ جُزْئهِ الأوَّلِ كاملاً، إذْ تجبُ فيهِ المراقبةُ، ومِن ثَــمَّ يُـحذفُ بعضُهُ وجوبًا. 5

أبحرُها: 6

يتولَّدُ مِن هذِهِ الدّائرةِ تسعةُ أبحرٍ، ستةٌ مستعملةٌ، وثلاثَةٌ مهملةٌ وأبحرُهَا هي: 8

أبحرُ هذهِ الدّائرةِ: 9

1- بحرُ السريعِ، وميزانُهُ: 10

2- بحرُ المتَّئدِ (المهمل)، وميزانُهُ: 11

3- بحرُ المنسردِ (المهمل)، وميزانُهُ: 12

4- بحرُ المنسرحِ، وميزانُهُ: 13

5- بحرُ الخفيفِ، وميزانُهُ: 14

6- بحرُ المضَارع، وميزانُهُ: 15

7- بحرُ المقتَضَب، وميزانُهُ: 16

8- بحرُ المجْتَثّ، وميزانُهُ: 17

9- بحرُ الـمُطَّرِدِ (المهمل)، وميزانُهُ: 18

طريقةُ فكِّ أبحرِ هذهِ الدّائرةِ: 19

أولاً: نبدأ بالسبب الخفيف الأوَّل في "مُسْتَفْعِلُنْ" المقابل للرَّقم[1] فيتولّدُ بحرُ السَّريع "مُسْتَفْعِلُنْ  مُسْتَفْعِلُنْ  مَفْعُولات". 21

ثانياً: نبدأ بالسَّببِ الخفيفِ الثّاني في "مُسْتَفْعِلُنْ" المقابلِ للرَّقم[3]، فيتولَّدُ بحرُ المتَّئد "المهمل"، "فَاعِلاتُنْ  فَاعِلاتُن  مُسْتَفْعِ لُنْ". 3

ثالثاً: نبدأ بالوتدِ المجموعِ في "مستفعلن" المقابلِ للرَّقم [5]، فيتولَّدُ بحرُ المنسرِدِ "المهمل"، "مَفَاعِيلُن مَفَاعِيلُن فَاعِ لاتُن". 5

رابعاً: نبدأ بالسَّببِ الخفيفِ الأوَّل في "مُسْتَفْعِلُنْ" الثَّانية المقابل للرَّقم [8]، فيتولَّدُ المنسرحُ، "مُسْتَفْعِلُنْ مَفْعُولات مُسْتَفْعِلُنْ". 7

خامساً: نبدأ بالسَّببِ الخفيفِ الثّاني في "مُسْتَفْعِلُنْ" الثّانية المقابل للرَّقمِ[10]، فيتولَّدُ الخفيفُ، "فَاعِلاتُن    مُسْتَفْعِ لُنْ    فَاعِلاتُن". 10

سادساً: نبدأ بالوتدِ المجموعِ في "مستفعلنُ" الثانية المقابل للرَّقم [12]، فيتولَّدُ المضارع، "مَفاعِيلُن فَاع لاتُن مَفَاعِيلُن". 12

سابعاً: نبدأ بالسَّببِ الخفيفِ الأوَّلِ في "مَفْعُولات" المقابل للرَّقم[15]، فيتولَّدُ المقتضَبُ، "مَفْعُولات  مُسْتَفْعِلُن  مُسْتَفْعِلُن". 14

ثامناً: نبدأ بالسَّببِ الخفيفِ الثّاني في "مَفْعُولات" المقابل للرَّقم[17]، فيتولَّدُ المجْتَثُّ، "مُسْتَفْعِ لُنْ   فَاعِلاتُنْ   فَاعِلاتُنْ". 16

تاسعاً: نبدأ بالوتَدِ المفروقِ في "مفعولات" المقابل للرَّقم[19]، فيتولَّدُ بحر المطَّرِدِ ، "فَاعِ لاتُنْ   مَفَاعِيلُن    مَفَاعِيلُن". 18
